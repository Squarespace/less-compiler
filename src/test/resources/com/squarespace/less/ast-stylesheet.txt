:PROPERTIES
parser=STYLESHEET

:LESS
@charset 'utf-8';

@media screen {
  .foo {
    color: red;
  }
  .bar {
    color: blue;
  }
}

.baz {
  font-size: 10px;
}

:REPR
[STYLESHEET, 1, [
  [DIRECTIVE, '@charset', [QUOTED, 0, false, [
    [ANONYMOUS, 'utf-8']
  ]]],
  [MEDIA, [FEATURES, [
    [EXPRESSION, [
      [KEYWORD, 'screen']
    ]]
  ]], [
    [RULESET, [SELECTORS, [
      [SELECTOR, [
        [TEXT_ELEMENT, ' ', '.foo']
      ]]
    ]], [
      [RULE, [PROPERTY, 'color'], [COLOR, 255, 0, 0, 1.0], false]
    ]],
    [RULESET, [SELECTORS, [
      [SELECTOR, [
        [TEXT_ELEMENT, ' ', '.bar']
      ]]
    ]], [
      [RULE, [PROPERTY, 'color'], [COLOR, 0, 0, 255, 1.0], false]
    ]]
  ]],
  [RULESET, [SELECTORS, [
    [SELECTOR, [
      [TEXT_ELEMENT, ' ', '.baz']
    ]]
  ]], [
    [RULE, [PROPERTY, 'font-size'], [DIMENSION, 10, 'px'], false]
  ]]
]]
